## Block Guests ##
ip access-list extended GUEST_BLOCK
 deny ip 192.168.50.0 0.0.0.255 192.168.0.0 0.0.255.255
 permit ip any any

interface vlan 50
 ip access-group GUEST_BLOCK in

## SSH Server ##
Standard IP access list SSH_IN
    10 permit 192.168.30.0 0.0.0.255

## VLAN 10 ##
Extended IP access list HR_IN
    5 permit udp any eq bootpc any eq bootps
    10 permit udp 192.168.10.0 0.0.0.255 host 192.168.40.4
    15 permit udp 192.168.10.0 0.0.0.255 host 192.168.40.2 eq domain
    20 permit tcp 192.168.10.0 0.0.0.255 host 192.168.40.1 eq ftp
Extended IP access list HR_OUT
    5 permit udp host 192.168.40.4 192.168.10.0 0.0.0.255
    10 permit udp host 192.168.40.2 eq domain 192.168.10.0 0.0.0.255
    20 permit tcp host 192.168.40.1 eq ftp 192.168.10.0 0.0.0.255

## VLAN 20CL ##

    5 permit udp host 192.168.40.4 192.168.20.0 0.0.0.255
    10 permit udp host 192.168.40.2 eq domain 192.168.20.0 0.0.0.255
    20 permit tcp host 192.168.40.3 eq 443 192.168.20.0 0.0.0.255
Extended IP access list SALES_IN
    5 permit udp any eq bootpc any eq bootps
    10 permit udp 192.168.20.0 0.0.0.255 host 192.168.40.4
    15 permit udp 192.168.20.0 0.0.0.255 host 192.168.40.2 eq domain
    20 permit tcp 192.168.20.0 0.0.0.255 host 192.168.40.3 eq 443
